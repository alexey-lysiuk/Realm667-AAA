actor ZDSClip : CustomInventory replaces Clip
{
    Inventory.PickupMessage "You picked up 5 souls"

    States
    {
    Spawn:
        ZDSS ABCB 6 BRIGHT
        loop
    Pickup:
        TNT1 A 0 A_GiveInventory("ZDSSoul", 5)
        stop
    }
}
